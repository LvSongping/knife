88  /*
89  * Load agent library If isAbsolute is true then the agent library is the
90  * absolute path to the library and thus will not be expanded in the target
91  * VM. if isAbsolute is false then the agent library is just a library name
92  * and it will be expended in the target VM.
93  */
94  private void loadAgentLibrary(String agentLibrary, boolean isAbsolute,
95  String options) throws AgentLoadException,
96  AgentInitializationException, IOException {
97  InputStream in = execute("load", agentLibrary, isAbsolute ? "true"
98  : "false", options);
99  try {
100  int result = readInt(in);
101  if (result != 0) {
102  throw new AgentInitializationException("Agent_OnAttach failed",
103  result);
104  }
105  } finally {
106  in.close();
107 
108  }
109  }

129  /*
130  * Load JPLIS agent which will load the agent JAR file and invoke the
131  * agentmain method.
132  */
133  public void loadAgent(String agent, String options)
134  throws AgentLoadException, AgentInitializationException,
135  IOException {
136  String args = agent;
137  if (options != null) {
138  args = args + "=" + options;
139  }
140  try {
141  loadAgentLibrary("instrument", args);
142  } catch (AgentLoadException x) {
143  throw new InternalError(
144  "instrument library is missing in target VM");
145  } catch (AgentInitializationException x) {
146  /*
147  * Translate interesting errors into the right exception and message
148  * (FIXME: create a better interface to the instrument
149  * implementation so this isn't necessary)
150  */
151  int rc = x.returnValue();
152  switch (rc) {
153  case JNI_ENOMEM:
154  throw new AgentLoadException("Insuffient memory");
155  case ATTACH_ERROR_BADJAR:
156  throw new AgentLoadException(
157  "Agent JAR not found or no Agent-Class attribute");
158  case ATTACH_ERROR_NOTONCP:
159  throw new AgentLoadException(
160  "Unable to add JAR file to system class path");
161  case ATTACH_ERROR_STARTFAIL:
162  throw new AgentInitializationException(
163  "Agent JAR loaded but agent failed to initialize");
164  default:
165  throw new AgentLoadException(
166  "Failed to load agent - unknown reason: " + rc);
167  }
168  }
169  }

264  int readInt(InputStream in) throws IOException {
265  StringBuilder sb = new StringBuilder();
266 
267  // read to \n or EOF
268  int n;
269  byte buf[] = new byte[1];
270  do {
271  n = in.read(buf, 0, 1);
272  if (n > 0) {
273  char c = (char) buf[0];
274  if (c == '\n') {
275  break; // EOL found
276  } else {
277  sb.append(c);
278  }
279  }
280  } while (n > 0);
281 
282  if (sb.length() == 0) {
283  throw new IOException("Premature EOF");
284  }
285 
286  int value;
287  try {
288  value = Integer.parseInt(sb.toString());
289  } catch (NumberFormatException x) {
290  throw new IOException("Non-numeric value found - int expected");
291  }
292  return value;
293  }

295  // -- attach timeout support

297  private static long defaultAttachTimeout = 5000;
298  private volatile long attachTimeout;
299 
300  /*
301  * Return attach timeout based on the value of the
302  * sun.tools.attach.attachTimeout property, or the default timeout if the
303  * property is not set to a positive value.
304  */
305  long attachTimeout() {
306  if (attachTimeout == 0) {
307  synchronized (this) {
308  if (attachTimeout == 0) {
309  try {
310  String s = System
311  .getProperty("sun.tools.attach.attachTimeout");
312  attachTimeout = Long.parseLong(s);
313  } catch (SecurityException se) {
314  } catch (NumberFormatException ne) {
315  }
316  if (attachTimeout <= 0) {
317  attachTimeout = defaultAttachTimeout;
318  }
319  }
320  }
321  }
322  return attachTimeout;
323  }
