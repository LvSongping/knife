54  InputStream execute(String cmd, Object... args) throws AgentLoadException,
55  IOException {
56  assert args.length <= 3; // includes null
57 
58  // did we detach?
59  String p;
60  synchronized (this) {
61  if (this.path == null) {
62  throw new IOException("Detached from target VM");
63  }
64  p = this.path;
65  }
66 
67  // create UNIX socket
68  int s = socket();
69 
70  // connect to target VM
71  try {
72  connect(s, p);
73  } catch (IOException x) {
74  close(s);
75  throw x;
76  }
77 
78  IOException ioe = null;
79 
80  // connected - write request
81  // <ver> <cmd> <args...>
82  try {
83  writeString(s, PROTOCOL_VERSION);
84  writeString(s, cmd);
85 
86  for (int i = 0; i < 3; i++) {
87  if (i < args.length && args[i] != null) {
88  writeString(s, (String) args[i]);
89  } else {
90  writeString(s, "");
91  }
92  }
93  } catch (IOException x) {
94  ioe = x;
95  }
96 
97  // Create an input stream to read reply
98  SocketInputStream sis = new SocketInputStream(s);
99 
100  // Read the command completion status
101  int completionStatus;
102  try {
103  completionStatus = readInt(sis);
104  } catch (IOException x) {
105  sis.close();
106  if (ioe != null) {
107  throw ioe;
108  } else {
109  throw x;
110  }
111  }
112 
113  if (completionStatus != 0) {
114  sis.close();
115 
116  // In the event of a protocol mismatch then the target VM
117  // returns a known error so that we can throw a reasonable
118  // error.
119  if (completionStatus == ATTACH_ERROR_BADVERSION) {
120  throw new IOException("Protocol mismatch with target VM");
121  }
122 
123  // Special-case the "load" command so that the right exception is
124  // thrown.
125  if (cmd.equals("load")) {
126  throw new AgentLoadException("Failed to load agent library");
127  } else {
128  throw new IOException("Command failed in target VM");
129  }
130  }
131 
132  // Return the input stream so that the command output can be read
133  return sis;
134  }

136  int readInt(InputStream inputstream) throws IOException {
137  StringBuilder stringbuilder = new StringBuilder();
138  byte abyte0[] = new byte[1];
139  int i;
140  do {
141  i = inputstream.read(abyte0, 0, 1);
142  if (i <= 0)
143  continue;
144  char c = (char) abyte0[0];
145  if (c == '\n')
146  break;
147  stringbuilder.append(c);
148  } while (i > 0);
149  if (stringbuilder.length() == 0)
150  throw new IOException("Premature EOF");
151  int j;
152  try {
153  j = Integer.parseInt(stringbuilder.toString());
154  } catch (NumberFormatException numberformatexception) {
155  throw new IOException("Non-numeric value found - int expected");
156  }
157  return j;
158  }

196  // Return the socket file for the given process.
197  // Checks working directory of process for .java_pid<pid>. If not
198  // found it looks in /tmp.
199  private String findSocketFile(int pid) {
200  // First check for a .java_pid<pid> file in the working directory
201  // of the target process
202  String fn = ".java_pid" + pid;
203  String path = "/proc/" + pid + "/cwd/" + fn;
204  File f = new File(path);
205  if (!f.exists()) {
206  // Not found, so try /tmp
207  path = "/tmp/" + fn;
208  f = new File(path);
209  if (!f.exists()) {
210  return null; // not found
211  }
212  }
213  return path;
214  }

216  // On Solaris/Linux a simple handshake is used to start the attach mechanism
217  // if not already started. The client creates a .attach_pid<pid> file in the
218  // target VM's working directory (or /tmp), and the SIGQUIT handler checks
219  // for the file.
220  private File createAttachFile(int pid) throws IOException {
221  String fn = ".attach_pid" + pid;
222  String path = "/proc/" + pid + "/cwd/" + fn;
223  File f = new File(path);
224  try {
225  f.createNewFile();
226  } catch (IOException x) {
227  path = "/tmp/" + fn;
228  f = new File(path);
229  f.createNewFile();
230  }
231  return f;
232  }
