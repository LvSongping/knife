91  private void lsField(Args args) {
92  ClassOrObject target = getTarget(args);
93  if (target.isNotFound()) {
94  Agent.sendResult(ResultHelper.newErrorResult("not found!"));
95  return;
96  }
97  List<FieldInfo> fieldInfos = new ArrayList<FieldInfo>();
98  if (target.getClazz() != null) {
99  Map<Field, Object> fieldMap = NativeHelper
100  .getStaticFieldValues(target.getClazz());
101  for (Entry<Field, Object> entry : fieldMap.entrySet()) {
102  FieldInfo info = new FieldInfo();
103  info.setStatic(true);
104  info.setName(entry.getKey().getName());
105  if (entry.getValue() != null) {
106  ObjectInfo fValue = new ObjectInfo();
107  fValue.setObjectId(ObjectRecordService.getInstance().toId(
108  entry.getValue()));
109  fValue.setValueString(toString(args, entry.getValue()));
110  info.setValue(fValue);
111  }
112  fieldInfos.add(info);
113  }
114  }
115  if (target.getObj() != null) {
116  Map<Field, Object> fieldMap = NativeHelper.getFieldValues(target
117  .getObj());
118  for (Entry<Field, Object> entry : fieldMap.entrySet()) {
119  FieldInfo info = new FieldInfo();
120  info.setStatic(false);
121  info.setName(entry.getKey().getName());
122  if (entry.getValue() != null) {
123  ObjectInfo fValue = new ObjectInfo();
124  fValue.setObjectId(ObjectRecordService.getInstance().toId(
125  entry.getValue()));
126  fValue.setValueString(toString(args, entry.getValue()));
127  info.setValue(fValue);
128  }
129  fieldInfos.add(info);
130  }
131  }
132  ClassFieldInfo info = new ClassFieldInfo();
133  info.setFields(fieldInfos.toArray(new FieldInfo[fieldInfos.size()]));
134  Agent.sendResult(ResultHelper.newResult(info));
